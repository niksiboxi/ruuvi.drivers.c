534 static bool sensor_returns_continuous_data (const rd_sensor_t * const DUT)
535 {
536  rd_status_t err_code = RD_SUCCESS;
537  uint8_t mode = RD_SENSOR_CFG_SLEEP;
538  float old_values[MAX_SENSOR_PROVIDED_FIELDS];
539  float new_values[MAX_SENSOR_PROVIDED_FIELDS];
540  rd_sensor_data_t old_data = {.fields = DUT->provides,
541  .data = old_values
542  };
543  rd_sensor_data_t new_data = {.fields = DUT->provides,
544  .data = new_values
545  };
546  err_code = DUT->mode_set (&mode);
547  err_code |= DUT->data_get (&old_data);
548  ri_delay_ms (2); // wait 2 ms to ensure timestamp is not changed.
549  err_code |= DUT->data_get (&new_data);
550  err_code = DUT->mode_set (&mode);
551  uint8_t samplerate = RD_SENSOR_CFG_MAX;
552  err_code |= DUT->samplerate_set (&samplerate);
553  mode = RD_SENSOR_CFG_CONTINUOUS;
554  err_code |= DUT->mode_set (&mode);
555  uint32_t interval = (1000 / (samplerate + 1));
556  ri_delay_ms (2 * interval);
557  err_code |= DUT->data_get (&old_data);
558  int retries = 0;
559 
560  for (; retries < MAX_RETRIES; retries++)
561  {
562  ri_delay_ms (2 * interval);
563  new_data.valid.bitfield = 0;
564  new_data.timestamp_ms = RD_SENSOR_INVALID_TIMSTAMP;
565  err_code |= DUT->data_get (&new_data);
566 
567  if (old_data.timestamp_ms == new_data.timestamp_ms || RD_SUCCESS != err_code)
568  {
569  if (RD_STATUS_MORE_AVAILABLE == err_code)
570  {
571  do
572  {
573  err_code |= DUT->data_get (&new_data);
574  } while (RD_STATUS_MORE_AVAILABLE == err_code);
575 
576  continue;
577  }
578  else
579  {
580  return true;
581  }
582  }
583 
584  if (value_has_changed (&old_data, &new_data)) { break; }
585  }
586 
587  if (MAX_RETRIES == retries)
588  {
589  return true;
590  }
591 
592  return false;
593 }

705 static rd_status_t test_sensor_fifo_enable (const rd_sensor_t * DUT)
706 {
707  DUT->fifo_enable (true);
708  rd_sensor_configuration_t config = {0};
709  config.samplerate = RD_SENSOR_CFG_MAX;
710  config.mode = RD_SENSOR_CFG_CONTINUOUS;
711  DUT->configuration_set (DUT, &config);
712  fifo_int = false;
713  ri_delay_ms (100);
714  rd_sensor_data_t old;
715  float old_values[MAX_SENSOR_PROVIDED_FIELDS];
716  old.data = old_values;
717  old.fields.bitfield = DUT->provides.bitfield;
718  size_t num_samples = MAX_FIFO_DEPTH;
719  rd_sensor_data_t data[MAX_FIFO_DEPTH] = { 0 };
720  float values[num_samples][MAX_SENSOR_PROVIDED_FIELDS];
721 
722  for (size_t ii = 0; ii < num_samples; ii++)
723  {
724  data[ii].data = values[ii];
725  data[ii].fields.bitfield = DUT->provides.bitfield;
726  }
727 
728  bool valid_data = false;
729  // Wait for FIFO interrupt
730  uint32_t timeout = 0;
731 
732  while ( (!fifo_int) && (timeout < 1000000U))
733  {
734  timeout += 10;
735  ri_delay_us (10);
736  }
737 
738  if (timeout >= 1000000U)
739  {
740  return RD_ERROR_TIMEOUT;
741  }
742 
743  DUT->fifo_read (&num_samples, data);
744 
745  if (10U > num_samples)
746  {
747  return RD_ERROR_SELFTEST;
748  }
749 
750  // Check that FIFO has new values
751  value_has_changed (&old, & (data[0]));
752 
753  for (size_t iii = 1; iii < num_samples; iii++)
754  {
755  if (value_has_changed (&old, & (data[iii])))
756  {
757  valid_data = true;
758  break;
759  }
760  }
761 
762  return (valid_data) ? false : true;
763 }
