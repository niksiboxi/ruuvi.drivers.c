478 rd_status_t ri_bme280_mode_set (uint8_t * mode)
479 {
480  if (NULL == mode) { return RD_ERROR_NULL; }
481 
482  rd_status_t err_code = RD_SUCCESS;
483  uint8_t current_mode = RD_SENSOR_CFG_SLEEP;
484 
485  switch (*mode)
486  {
487  case RD_SENSOR_CFG_SLEEP:
488  err_code = BME_TO_RUUVI_ERROR (bme280_set_sensor_mode (BME280_SLEEP_MODE, &dev));
489  break;
490 
491  case RD_SENSOR_CFG_SINGLE:
492  // Do nothing if sensor is in continuous mode
493  ri_bme280_mode_get (&current_mode);
494 
495  if (RD_SENSOR_CFG_CONTINUOUS == current_mode)
496  {
497  *mode = RD_SENSOR_CFG_CONTINUOUS;
498  return RD_ERROR_INVALID_STATE;
499  }
500 
501  err_code = BME_TO_RUUVI_ERROR (bme280_set_sensor_mode (BME280_FORCED_MODE, &dev));
502  // We assume that dev struct is in sync with the state of the BME280 and underlying interface
503  // which has the number of settings as 2^OSR is not changed.
504  // We also assume that each element runs same OSR
505  uint8_t samples = 1U << (dev.settings.osr_h - 1U);
506  ri_delay_ms (bme280_max_meas_time (samples));
507  tsample = rd_sensor_timestamp_get();
508  // BME280 returns to SLEEP after forced sample
509  *mode = RD_SENSOR_CFG_SLEEP;
510  break;
511 
512  case RD_SENSOR_CFG_CONTINUOUS:
513  err_code = BME_TO_RUUVI_ERROR (bme280_set_sensor_mode (BME280_NORMAL_MODE, &dev));
514  break;
515 
516  default:
517  err_code = RD_ERROR_INVALID_PARAM;
518  break;
519  }
520 
521  return err_code;
522 }
