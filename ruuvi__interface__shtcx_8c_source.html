252 rd_status_t ri_shtcx_mode_set (uint8_t * mode)
253 {
254  if (NULL == mode) { return RD_ERROR_NULL; }
255 
256  // Enter sleep by default and by explicit sleep commmand
257  if (RD_SENSOR_CFG_SLEEP == *mode || RD_SENSOR_CFG_DEFAULT == *mode)
258  {
259  m_autorefresh = false;
260  *mode = RD_SENSOR_CFG_SLEEP;
261  return RD_SUCCESS;
262  }
263 
264  if (RD_SENSOR_CFG_SINGLE == *mode)
265  {
266  // Do nothing if sensor is in continuous mode
267  uint8_t current_mode;
268  ri_shtcx_mode_get (&current_mode);
269 
270  if (RD_SENSOR_CFG_CONTINUOUS == current_mode)
271  {
272  *mode = RD_SENSOR_CFG_CONTINUOUS;
273  // Start first measurement
274  shtc1_measure();
275  return RD_ERROR_INVALID_STATE;
276  }
277 
278  // Enter sleep after measurement
279  m_autorefresh = false;
280  *mode = RD_SENSOR_CFG_SLEEP;
281  m_tsample = rd_sensor_timestamp_get();
282  return SHTCX_TO_RUUVI_ERROR (shtc1_measure_blocking_read (&m_temperature, &m_humidity));
283  }
284 
285  if (RD_SENSOR_CFG_CONTINUOUS == *mode)
286  {
287  m_autorefresh = true;
288  return RD_SUCCESS;
289  }
290 
291  return RD_ERROR_INVALID_PARAM;
292 }

311 rd_status_t ri_shtcx_data_get (rd_sensor_data_t * const
312  p_data)
313 {

315 
316  rd_status_t err_code = RD_SUCCESS;
317 
318  if (m_autorefresh)
319  {
320  /* Sensor sleep clears measured values, blocking read required.
321  // read sensor values
322  err_code |= SHTCX_TO_RUUVI_ERROR(shtc1_read(&m_temperature, &m_humidity));
323  // Start next measurement

328  m_tsample = rd_sensor_timestamp_get();
329  }
330 
331  if (RD_SUCCESS == err_code && RD_UINT64_INVALID != m_tsample)
332  {
333  rd_sensor_data_t d_environmental;
334  rd_sensor_data_fields_t env_fields = {.bitfield = 0};
335  float env_values[2];

344  rd_sensor_data_populate (p_data,
